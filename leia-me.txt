<script>
/* =====================================================
   PlacaLens — usa Claude Vision (claude-sonnet-4-20250514)
   para reconhecimento de placas brasileiras.
   Prompt especializado garante acerto altíssimo nos dois
   formatos: Mercosul (LLLNLNN) e Antiga (LLLNNNN).
   ===================================================== */

const $ = id => document.getElementById(id);

// ── state ──
let stream = null, cameraOn = false, photoMode = false;
let autoMode = false, autoRunning = false;
let records = [], sessionStart = Date.now();
let fCount = 0, fLast = Date.now();

// ── boot ──
(async function boot() {
  $('loaderTxt').textContent = 'PRONTO';
  $('apiDot').className = 'status-dot ok';
  $('apiLabel').textContent = 'CLAUDE VISION';
  await delay(600);
  $('loader').classList.add('gone');
  setInterval(updateSession, 1000);
  setInterval(updateFPS, 800);
  setupUpload();
  startCamera();
})();

// ── camera ──
async function startCamera() {
  try {
    stream = await navigator.mediaDevices.getUserMedia({
      video: { facingMode: { ideal: 'environment' }, width: { ideal: 1920 }, height: { ideal: 1080 } }
    });
    $('video').srcObject = stream;
    await $('video').play();
    cameraOn = true; photoMode = false;
    $('photoPreview').style.display = 'none';
    $('video').style.display = 'block';
    $('photoStrip').classList.remove('show');
    $('btnCam').textContent = '\u25a0 PARAR';
    $('btnCam').classList.add('on');
    $('btnAuto').disabled = false;
    $('btnSnap').disabled = false;
    $('noCam').classList.remove('show');
    requestAnimationFrame(fpsLoop);
  } catch(e) {
    $('noCam').classList.add('show');
  }
}
function stopCamera() {
  if (stream) stream.getTracks().forEach(t => t.stop());
  stream = null; cameraOn = false;
  $('video').srcObject = null;
  stopAuto();
  $('btnCam').textContent = '\u25ba C\u00c2MERA';
  $('btnCam').classList.remove('on');
  $('btnAuto').disabled = true;
  $('btnSnap').disabled = true;
}
function toggleCamera() { cameraOn ? stopCamera() : startCamera(); }
function fpsLoop() { fCount++; if (cameraOn) requestAnimationFrame(fpsLoop); }
function updateFPS() { const n = Date.now(); fCount = 0; fLast = n; }

// ── AUTO-SCAN ──
function toggleAuto() { autoMode ? stopAuto() : startAuto(); }
function startAuto() {
  if (!cameraOn || photoMode) return;
  autoMode = true;
  $('btnAuto').textContent = '\u25a0 PARAR AUTO';
  $('btnAuto').classList.add('on');
  $('scanLine').classList.add('active');
  $('autoBadge').classList.add('show');
  setDetect('BUSCANDO PLACA...', 'searching');
  autoLoop();
}
function stopAuto() {
  autoMode = false; autoRunning = false;
  $('btnAuto').textContent = '\u25b6 AUTO-SCAN';
  $('btnAuto').classList.remove('on');
  $('scanLine').classList.remove('active');
  $('autoBadge').classList.remove('show');
  setDetect('', '');
}
async function autoLoop() {
  while (autoMode) {
    if (!autoRunning && cameraOn && $('video').readyState >= 2) {
      autoRunning = true;
      try {
        const b64 = frameToBase64(true); // low-res for speed
        const result = await callClaudeVision(b64, 'image/jpeg', true);
        if (result.plate && autoMode) {
          setDetect('\u2713 ' + fmtPlate(result.plate), 'found');
          $('reticleBox').classList.add('hit');
          setTimeout(() => $('reticleBox').classList.remove('hit'), 800);
          registerPlate(result.plate, result.type, result.confidence, result.raw, 'AUTO');
          toast('\u2713 AUTO: ' + fmtPlate(result.plate));
          autoRunning = false;
          await delay(4000); // cooldown after detection
          if (autoMode) setDetect('BUSCANDO PLACA...', 'searching');
        } else {
          setDetect('BUSCANDO PLACA...', 'searching');
        }
      } catch(e) { console.warn('auto:', e); }
      autoRunning = false;
    }
    await delay(1200); // scan interval
  }
}

// ── CLAUDE VISION API ──
// Usa prompt cirúrgico para extrair APENAS a placa, nos dois formatos.
// fast=true usa resolução menor e temperatura 0 para auto-scan
async function callClaudeVision(b64, mime, fast) {
  const prompt = fast
    ? `Look at this image. If you see a Brazilian license plate, respond with ONLY the plate characters (no spaces, no punctuation, no explanation). If no plate is visible or readable, respond with exactly: NONE\n\nBrazilian plates have 7 characters:\n- Mercosul: 3 letters + 1 digit + 1 letter + 2 digits (ex: ABC1D23)\n- Old format: 3 letters + 4 digits (ex: ABC1234)\n\nRespond with ONLY the 7 characters or NONE.`
    : `You are a Brazilian license plate reader. Analyze this image carefully.\n\nBrazilian license plates have exactly 7 characters:\n- Mercosul format: 3 letters, 1 digit, 1 letter, 2 digits — example: ABC1D23\n- Old format: 3 letters, 4 digits — example: ABC1234\n\nRules:\n1. Inspect the entire image for any license plate\n2. If you find a plate, return ONLY the 7 alphanumeric characters with no spaces, dashes, or punctuation\n3. Use uppercase letters only\n4. If you cannot find or read a plate, respond with exactly: NONE\n5. Do NOT guess or hallucinate — if uncertain, return NONE\n\nRespond with ONLY the 7-character plate code or NONE. Nothing else.`;

  const resp = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 30,
      messages: [{
        role: 'user',
        content: [{
          type: 'image',
          source: { type: 'base64', media_type: mime.startsWith('image/') ? mime : 'image/jpeg', data: b64 }
        }, {
          type: 'text', text: prompt
        }]
      }]
    })
  });

  if (!resp.ok) {
    const err = await resp.json().catch(() => ({}));
    throw new Error(err.error?.message || 'API error ' + resp.status);
  }

  const data = await resp.json();
  const raw = (data.content?.[0]?.text || '').trim().toUpperCase().replace(/[^A-Z0-9]/g, '');

  if (!raw || raw === 'NONE' || raw.length < 7) return { plate: null, raw: data.content?.[0]?.text || '' };

  // Extract 7-char plate from response
  const plate = extractPlate(raw);
  if (!plate) return { plate: null, raw: data.content?.[0]?.text || '' };

  const isMerc = /^[A-Z]{3}[0-9][A-Z][0-9]{2}$/.test(plate);
  return {
    plate,
    type: isMerc ? 'mercosul' : 'antiga',
    confidence: 95, // Claude Vision has very high confidence
    raw: data.content?.[0]?.text || plate
  };
}

function extractPlate(s) {
  const c = s.replace(/[^A-Z0-9]/g, '');
  // Mercosul: LLL N L NN
  let m = c.match(/[A-Z]{3}[0-9][A-Z][0-9]{2}/);
  if (m) return m[0];
  // Antiga: LLL NNNN
  m = c.match(/[A-Z]{3}[0-9]{4}/);
  if (m) return m[0];
  return null;
}

// ── CANVAS HELPERS ──
function frameToBase64(lowRes) {
  const v = $('video');
  const scale = lowRes ? 0.5 : 1.0;
  const c = document.createElement('canvas');
  c.width  = Math.round(v.videoWidth  * scale);
  c.height = Math.round(v.videoHeight * scale);
  c.getContext('2d').drawImage(v, 0, 0, c.width, c.height);
  const dataUrl = c.toDataURL('image/jpeg', lowRes ? 0.7 : 0.9);
  return dataUrl.split(',')[1];
}

function imageToBase64(imgEl, mimeType) {
  return new Promise(resolve => {
    const c = document.createElement('canvas');
    // Cap at 2048px to keep API payload reasonable
    const maxDim = 2048;
    let w = imgEl.naturalWidth, h = imgEl.naturalHeight;
    if (w > maxDim || h > maxDim) {
      const ratio = Math.min(maxDim / w, maxDim / h);
      w = Math.round(w * ratio); h = Math.round(h * ratio);
    }
    c.width = w; c.height = h;
    const ctx = c.getContext('2d');
    ctx.drawImage(imgEl, 0, 0, w, h);
    const mime = mimeType && mimeType.startsWith('image/') ? mimeType : 'image/jpeg';
    const dataUrl = c.toDataURL(mime === 'image/png' ? 'image/png' : 'image/jpeg', 0.92);
    resolve(dataUrl.split(',')[1]);
  });
}
